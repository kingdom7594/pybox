import numpy as np import matplotlib.pyplot as plt from datetime import datetime # 计算 Fibonacci 数列 def fibonacci(n): if n <= 1: return n return fibonacci(n-1) + fibonacci(n-2) # 绘制 spiral def draw_spiral(): angles = np.linspace(0, 4*np.pi, 100) r = np.linspace(1, 10, 100) plt.figure(figsize=(8, 6)) plt.polar(True) plt.plot(angles, r, 'b-', linewidth=2) plt.title(f' Spiral - {datetime.now().strftime("%Y-%m-%d")}') plt.show() if __name__ == "__main__": print("计算 Fibonacci...") for i in range(10): print(f"fib({i}) = {fibonacci(i)}") draw_spiral()

如何优化这段 Fibonacci 代码的性能？
可以使用记忆化（Memoization）来避免重复计算。这是一个经典的空间换时间优化：
from functools import lru_cache @lru_cache(maxsize=None) def fibonacci(n): if n <= 1: return n return fibonacci(n-1) + fibonacci(n-2)

这样可以将时间复杂度从 O(2ⁿ) 降到 O(n)。
